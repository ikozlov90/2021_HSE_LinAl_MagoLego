\section{Псевдообратные матрицы}

\subsection{Полезные формулы}

\textbf{Псевдообратная матрица}:

\begin{enumerate}


\item Если $A$ --- квадратная и обратимая \[ A^{+} = A^{-1}\]


\item Пусть столбцы $A$ линейно независимы. Тогда $A^* A$ обратима и \begin{equation} A^+ = (A^* A)^{-1} A^*.\end{equation}


\item Пусть строчки $A$ линейно независимы. Тогда $AA^*$ обратима и \begin{equation} A^+ = A^*(A A^*)^{-1}.\end{equation}


\item Пусть $A$ --- матрица размера $m \times n$ ранга $k$. Рассмотрим разложение полного ранга (скелетное разложение) \[A = FG,\] где  

\begin{itemize}

\item $F$ --- матрица размера $m \times k$ (столбцы линейно независимы)

\item $G$ --- матрица размера $k \times n$ (строки линейно независимыми).

\end{itemize}

 Тогда:\[ A^+ = G^+F^+\] Иными словами, \[A^+ = G^*(GG^*)^{-1}(F^*F)^{-1}F^*.\]




\end{enumerate}

\section{Многочлены Чебышёва}

\subsection{Теория}

\textbf{Многочлены Чебышёва 1го рода} $T_n(x)$ определяются рекуррентным соотношением: \[{\displaystyle {\begin{aligned}T_{0}(x)&=1\\T_{1}(x)&=x\\T_{n+1}(x)&=2x\,T_{n}(x)-T_{n-1}(x)~.\end{aligned}}}\]

На отрезке $[-1, 1]$ они задаются формулой \[ {\displaystyle T_{n}(x)=\cos {\big (}n\arccos(x){\big )}}.\] 

$T_n(x)$ --- это многочлен степени $n$ со старшим коэффициентом $2^{{n-1}}$, наименее отклоняющийся от нуля на отрезке $[-1, 1]$. 

Многочлены Чебышёва $T_n(x)$ ортогональны относительно скалярного произведения \[ \langle f, g \rangle = \int_{-1}^1 \frac{f(x) g(x)}{\sqrt{1-x^2}} dx. \] А именно \[ \langle T_m, T_n \rangle = \begin{cases} 0, \qquad m \not = n, \\
%
\displaystyle \frac{\pi}{2}, \qquad m = n \not = 0, \\
%
\pi, \qquad m = n =0. 
\end{cases}  \]  



\textbf{Многочлены Чебышёва 2го рода} $T_n(x)$ определяются рекуррентным соотношением: \[{\displaystyle {\begin{aligned}U_{0}(x)&=1\\U_{1}(x)&=2x\\U_{n+1}(x)&=2x\,U_{n}(x)-U_{n-1}(x)~.\end{aligned}}}\]
